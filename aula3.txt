CREATE TABLE DEPARTAMENTO(
ID_DEPART INTEGER,
NOME_DEPART VARCHAR2(40),
constraint DEPART_ID_PK PRIMARY KEY(ID_DEPART)
);

CREATE TABLE FUNCIONARIO(
ID_FUNC INTEGER,
NOME_FUNC VARCHAR2(40),
ENDERECO VARCHAR2(40),
CEP VARCHAR(9),
CIDADE VARCHAR(20),
UF CHAR(2),
TELEFONE VARCHAR(15),
ID_DEPART INTEGER NOT NULL
);

CREATE TABLE DEPENDENTE(
ID_DEP INTEGER,
NOME VARCHAR(40),
ID_FUNC INTEGER NOT NULL,
constraint DEPEND_ID_PK PRIMARY KEY(ID_DEP)
);
